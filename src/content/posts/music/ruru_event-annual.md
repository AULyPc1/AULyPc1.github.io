<script>
(function() {
  function scaleAllTables() {
    var tables = document.querySelectorAll('.scale-table');
    tables.forEach(function(table) {
      var wrapper = table.closest('.table-wrapper');
      if (!wrapper) return;
      var tableWidth = 800; // 表格原始宽度
      var wrapperWidth = wrapper.offsetWidth;
      var scale = Math.min(1, wrapperWidth / tableWidth);
      table.style.transform = 'scale(' + scale + ')';
      table.style.transformOrigin = 'left top';
      // 设置父容器高度为缩放后的表格高度，消除多余空白
      wrapper.style.height = (table.offsetHeight * scale) + 'px';
      wrapper.style.position = 'relative';
    });
  }
  window.addEventListener('resize', scaleAllTables);
  window.addEventListener('DOMContentLoaded', scaleAllTables);
})();
</script>
